A guessing game that I made with python

import random 

value = random.randint(1, 100)

guess = int(input("Guess a number between 1 and 100"))
while value != guess:
    if guess < value:
        print("too low")
        guess = int(input("Guess a number between 1 and 100"))
    elif guess > value:
        print("too high")
        guess = int(input("Guess a number between 1 and 100"))

print("correct!")
